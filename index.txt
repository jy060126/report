        function startWorldCup(num) {
            totalRounds = num;

        function showMatch() {
            const progress = ((matchIndex / 2) / (totalRounds / 2)) * 100;
            document.getElementById('progress-bar').style.width = progress + '%';
            const container = document.getElementById('game-container');
            const cardL = document.getElementById('card-left');
            const cardR = document.getElementById('card-right');
            container.classList.remove('select-anim-active', 'select-left-anim', 'select-right-anim');
            cardL.classList.remove('card-selected');
            cardR.classList.remove('card-selected');
            const left = currentMatches[matchIndex];
            const right = currentMatches[matchIndex + 1];
            document.getElementById('round-title').innerText = totalRounds === 2 ? "결승전" : `${totalRounds}강 (${(matchIndex/2)+1}/${totalRounds/2})`;
            document.getElementById('img-left').src = left.img;
            document.getElementById('name-left').innerText = left.name;
            document.getElementById('img-right').src = right.img;
            document.getElementById('name-right').innerText = right.name;
            isProcessing = false; 
            const undoBtn = document.getElementById('undo-btn');
            if (matchIndex === 0 || totalRounds === 2) {
                undoBtn.style.display = 'none';
            } else {
                undoBtn.style.display = 'block';
            }
            const resetBtn = document.getElementById('reset-btn');

            if (totalRounds === 2) {
                resetBtn.style.display = 'none';
            } else {
                resetBtn.style.display = 'block';
            }
        }

        // 1. 아이돌 선택 로직 수정
        function selectIdol(offset) {
            if (isProcessing) return; 
            isProcessing = true;
            
            history.push({
                currentMatches: [...currentMatches],
                winners: [...winners],
                matchIndex: matchIndex,
                totalRounds: totalRounds
            });

            const container = document.getElementById('game-container');
            container.classList.add('select-anim-active');
            
            if (offset === 0) {
                container.classList.add('select-left-anim');
            } else {
                container.classList.add('select-right-anim');
            }

            // 선택된 아이돌을 승자 명단에 추가
            winners.push(currentMatches[matchIndex + offset]);
            matchIndex += 2;

            setTimeout(() => {
                container.style.opacity = '0';
                
                setTimeout(() => {
                    // 1. 현재 라운드에 대결할 팀이 더 남아있는 경우
                    if (matchIndex < currentMatches.length) {
                        showMatch();
                        container.style.opacity = '1';
                    } 
                    // 2. 현재 라운드의 모든 대결이 끝난 경우
                    else {
                        // 승자가 단 한 명이라면? -> 최종 우승!
                        if (winners.length === 1) {
                            showFinalWinner(winners[0]);
                        } 
                        // 승자가 여러 명이라면? -> 다음 강으로 진행 (예: 4강 -> 결승)
                        else {
                            currentMatches = [...winners];
                            winners = [];
                            matchIndex = 0;
                            totalRounds = currentMatches.length;
                            showRoundIntro(totalRounds === 2 ? "결승" : totalRounds + "강");
                        }
                    }
                }, 300); // 화면 전환 대기 시간
            }, 600); // 애니메이션 지속 시간
        }

            matchBox.innerHTML = `
                <div class="card winner-card">
                    <img src="${winner.img}" alt="${winner.name}">
                    <div class="card-name">${winner.name}</div>
                </div>

            // 결과 화면 노출

                shareContainer.style.display = 'flex';
                shareContainer.style.opacity = '1';
                resetBtn.style.display = 'block';
                resetBtn.style.opacity = '1';
            }, 500);

            saveWinner(winner); // 명예의 전당 저장
        }   

        function drawConstellation(id, stars, lines) {
            const container = document.getElementById(id);
            stars.forEach(p => {
                const s = document.createElement('div');
                s.className = 'const-star';
                s.style.left = p.x + 'px'; s.style.top = p.y + 'px';
                container.appendChild(s);
            });
            lines.forEach(l => {
                const p1 = stars[l[0]], p2 = stars[l[1]];
                const line = document.createElement('div');
                line.className = 'const-line';
                const dist = Math.sqrt(Math.pow(p2.x - p1.x, 2) + Math.pow(p2.y - p1.y, 2));
                const angle = Math.atan2(p2.y - p1.y, p2.x - p1.x) * 180 / Math.PI;
                line.style.width = dist + 'px';
                line.style.left = (p1.x + 2) + 'px'; line.style.top = (p1.y + 2) + 'px';
                line.style.transform = `rotate(${angle}deg)`;
                container.appendChild(line);
            });
        }

        // 우승 데이터 저장 (이름과 이미지를 매칭하여 저장)
        function saveWinner(winner) {
            let hallData = JSON.parse(localStorage.getItem('worldcup_hall_of_fame') || '{}');

            if (hallData[winner.name]) {
                hallData[winner.name].count += 1;
            } else {
                hallData[winner.name] = {
                    count: 1,
                    img: winner.img
                };
            }

            localStorage.setItem('worldcup_hall_of_fame', JSON.stringify(hallData));
        }

        // 명예의 전당 화면 표시 로직
        function displayHall() {
            const hallContent = document.getElementById('hall-content');
            const hallData = JSON.parse(localStorage.getItem('worldcup_hall_of_fame') || '{}');

            // 객체를 배열로 변환 후 우승 횟수 내림차순 정렬
            const sortedList = Object.entries(hallData).sort((a, b) => b[1].count - a[1].count);

            if (sortedList.length === 0) {
                hallContent.innerHTML = "<p style='grid-column: 1/-1; text-align: center; font-size: 1.5rem; opacity: 0.5;'>아직 우승 기록이 없습니다.</p>";
                return;
            }

            hallContent.innerHTML = sortedList.map(([name, data]) => `
                <div class="hall-card">
                    <img src="${data.img}" alt="${name}">
                    <div class="hall-info">
                        <div class="hall-name">${name}</div>
                        <div class="hall-count">누적 우승: ${data.count}회</div>
                    </div>
                </div>
            `).join('');
        }

        let startTime;
        let timerInterval = null;
        let elapsedTime = 0;

        // 2. 스톱워치 시작 함수
        function startStopwatch() {
            // 이미 타이머가 돌아가고 있다면 중복 실행하지 않음
            if (timerInterval) return; 

            startTime = Date.now() - elapsedTime;
            timerInterval = setInterval(updateDisplay, 10); // 10ms마다 updateDisplay 실행
            console.log("스톱워치 시작됨"); 
        }

        // 3. 화면 업데이트 함수
        function updateDisplay() {
            elapsedTime = Date.now() - startTime;
            
            const minutes = Math.floor(elapsedTime / 60000);
            const seconds = Math.floor((elapsedTime % 60000) / 1000);
            const milliseconds = Math.floor((elapsedTime % 1000) / 10);

            // 00:00.00 형식으로 맞추기
            const m = String(minutes).padStart(2, '0');
            const s = String(seconds).padStart(2, '0');
            const ms = String(milliseconds).padStart(2, '0');

            const display = document.getElementById('stopwatch-display');
            if (display) {
                display.innerText = `${m}:${s}.${ms}`;
            }
        }

        // 4. 스톱워치 정지 함수
        function stopStopwatch() {
            if (timerInterval) {
                clearInterval(timerInterval);
                timerInterval = null;
                console.log("스톱워치 정지됨");
            }
        }
